import "jsr:@supabase/functions-js/edge-runtime.d.ts"; import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization,apikey,content-type", "Access-Control-Allow-Methods": "POST,OPTIONS", "Content-Type": "application/json" }; Deno.serve(async (req) => { if (req.method === "OPTIONS") { return new Response(null, { status: 204, headers: corsHeaders }); } try { const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); const body = await req.json(); const app_id = String(body.app_id || ""); const plano_id = String(body.plano_id || ""); const mac = String(body.mac || ""); const ip = String(body.ip || ""); const hostname = String(body.hostname || ""); if (!app_id) throw new Error("APP_ID não informado."); if (!plano_id) throw new Error("Plano não informado."); console.log("APP:", app_id); console.log("PLANO:", plano_id); // ========================================== // CONFIGURAÇÃO DO HOTSPOT // ========================================== const { data: hotspot, error: erroHotspot } = await supabase .from("hotspot_config") .select("*") .eq("app_id", app_id) .single(); if (erroHotspot || !hotspot) throw new Error("Hotspot não encontrado."); // ========================================== // CONFIGURAÇÃO DA EMPRESA // ========================================== const { data: empresa, error: erroEmpresa } = await supabase .from("config_site") .select("*") .eq("app_id", app_id) .single(); if (erroEmpresa || !empresa) throw new Error("Empresa não encontrada."); const accessToken = empresa.mercado_pago_token || empresa.access_token || empresa.mp_access_token; if (!accessToken) throw new Error("Mercado Pago não configurado."); console.log("Empresa:", empresa.nome_empresa); // ========================================== // PLANO // ========================================== const { data: plano, error: erroPlano } = await supabase .from("hotspot_planos") .select("*") .eq("id", plano_id) .eq("app_id", app_id) .single(); if (erroPlano || !plano) throw new Error("Plano não encontrado."); console.log("Plano:", plano.nome); // ========================================== // CRIA O REGISTRO DO PAGAMENTO // ========================================== const { data: pagamento, error: erroPagamento } = await supabase .from("hotspot_pagamentos") .insert({ app_id, plano_id, valor: Number(plano.valor), status: "pendente", payment_status: "pending", cliente_mac: mac, cliente_ip: ip, hostname, plano_nome: plano.nome, tempo_minutos: plano.tempo_minutos, created_at: new Date().toISOString() }) .select() .single(); if (erroPagamento || !pagamento) { console.error(erroPagamento); throw new Error("Erro ao criar pagamento."); } console.log("Pagamento:", pagamento.id); // ========================================== // REFERÊNCIA EXTERNA // ========================================== const externalReference = String(pagamento.id); // ========================================== // MONTA O PIX // ========================================== const payload = { transaction_amount: Number(plano.valor), description: `Hotspot - ${plano.nome}`, payment_method_id: "pix", external_reference: externalReference, notification_url: "https://pdajixsoowcyhnjwhgpc.supabase.co/functions/v1/gerar-pix-hotspot", payer: { email: "cliente@ffhotspot.com", first_name: "Cliente", last_name: "Hotspot" } }; console.log("Payload MP"); console.log(payload); // ========================================== // MERCADO PAGO // ========================================== const mpResponse = await fetch( "https://api.mercadopago.com/v1/payments", { method: "POST", headers: { Authorization: `Bearer ${accessToken}`, "Content-Type": "application/json", "X-Idempotency-Key": crypto.randomUUID() }, body: JSON.stringify(payload) } ); const mp = await mpResponse.json(); console.log("Resposta Mercado Pago:"); console.log(mp); if (!mpResponse.ok) { console.error(mp); await supabase .from("hotspot_pagamentos") .update({ status: "erro", payment_status: "error", payload: mp, updated_at: new Date().toISOString() }) .eq("id", pagamento.id); throw new Error( mp.message || mp.error || "Erro ao gerar PIX." ); } // ========================================== // DADOS PIX // ========================================== const qrCode = mp.point_of_interaction ?.transaction_data ?.qr_code || ""; const qrCodeBase64 = mp.point_of_interaction ?.transaction_data ?.qr_code_base64 || ""; const paymentId = String(mp.id); const paymentStatus = mp.status || "pending"; console.log("PIX criado:", paymentId); // ========================================== // ATUALIZA PAGAMENTO // ========================================== const { error: erroAtualizar } = await supabase .from("hotspot_pagamentos") .update({ payment_id: paymentId, payment_status: paymentStatus, status: paymentStatus, qr_code: qrCode, qr_code_base64: qrCodeBase64, payload: mp, updated_at: new Date().toISOString() }) .eq("id", pagamento.id); if (erroAtualizar) { console.error(erroAtualizar); throw new Error("Erro ao salvar pagamento."); } console.log("Pagamento atualizado."); // ========================================== // RETORNO // ========================================== return new Response( JSON.stringify({ success: true, payment_id: paymentId, external_reference: externalReference, status: paymentStatus, valor: Number(plano.valor), plano: { id: plano.id, nome: plano.nome, descricao: plano.descricao, tempo_minutos: plano.tempo_minutos }, qr_code: qrCode, qr_code_base64: qrCodeBase64 }), { status: 200, headers: corsHeaders } ); } catch (err) { console.error("ERRO HOTSPOT:"); console.error(err); return new Response( JSON.stringify({ success: false, error: err instanceof Error ? err.message : String(err) }), { status: 500, headers: corsHeaders } ); } });
